```ts
// src/pricing/models.ts

export interface ModelPricing {
  input_per_mtok: number;
  cache_create_per_mtok: number;
  cache_read_per_mtok: number;
  output_per_mtok: number;
}

export const PRICING_VERSION = '2026-05-22';

// USD per million tokens. Sourced from public Anthropic / OpenAI pricing pages
// as of PRICING_VERSION. Cache rates follow Anthropic's published 1.25x / 0.1x
// multipliers; for OpenAI, cached input is half the input rate.
const PRICING: Record<string, ModelPricing> = {
  // Anthropic — Claude 4 family
  'claude-opus-4-7':    { input_per_mtok: 15.00, cache_create_per_mtok: 18.75, cache_read_per_mtok: 1.50, output_per_mtok: 75.00 },
  'claude-opus-4-6':    { input_per_mtok: 15.00, cache_create_per_mtok: 18.75, cache_read_per_mtok: 1.50, output_per_mtok: 75.00 },
  'claude-sonnet-4-6':  { input_per_mtok:  3.00, cache_create_per_mtok:  3.75, cache_read_per_mtok: 0.30, output_per_mtok: 15.00 },
  'claude-sonnet-4-5':  { input_per_mtok:  3.00, cache_create_per_mtok:  3.75, cache_read_per_mtok: 0.30, output_per_mtok: 15.00 },
  'claude-haiku-4-5':   { input_per_mtok:  0.80, cache_create_per_mtok:  1.00, cache_read_per_mtok: 0.08, output_per_mtok:  4.00 },
  // Anthropic — legacy
  'claude-3-7-sonnet':  { input_per_mtok:  3.00, cache_create_per_mtok:  3.75, cache_read_per_mtok: 0.30, output_per_mtok: 15.00 },
  'claude-3-5-sonnet':  { input_per_mtok:  3.00, cache_create_per_mtok:  3.75, cache_read_per_mtok: 0.30, output_per_mtok: 15.00 },
  'claude-3-5-haiku':   { input_per_mtok:  0.80, cache_create_per_mtok:  1.00, cache_read_per_mtok: 0.08, output_per_mtok:  4.00 },
  'claude-3-opus':      { input_per_mtok: 15.00, cache_create_per_mtok: 18.75, cache_read_per_mtok: 1.50, output_per_mtok: 75.00 },
};

const FALLBACK_BY_TIER = {
  opus:    PRICING['claude-opus-4-7'],
  sonnet:  PRICING['claude-sonnet-4-6'],
  haiku:   PRICING['claude-haiku-4-5'],
  unknown: PRICING['claude-sonnet-4-6'],
};

export interface PricingLookup {
  pricing: ModelPricing;
  fallback: boolean;
  matched_as: string;
}

export function pricingFor(modelId: string): PricingLookup {
  if (PRICING[modelId]) {
    return { pricing: PRICING[modelId], fallback: false, matched_as: modelId };
  }
  const m = modelId.toLowerCase();
  if (m.includes('opus'))   return { pricing: FALLBACK_BY_TIER.opus,    fallback: true, matched_as: 'opus-tier' };
  if (m.includes('haiku'))  return { pricing: FALLBACK_BY_TIER.haiku,   fallback: true, matched_as: 'haiku-tier' };
  if (m.includes('sonnet')) return { pricing: FALLBACK_BY_TIER.sonnet,  fallback: true, matched_as: 'sonnet-tier' };
  return { pricing: FALLBACK_BY_TIER.unknown, fallback: true, matched_as: 'unknown-fallback' };
}

export function listKnownModels(): string[] {
  return Object.keys(PRICING);
}
```

### Task 2: Cost computation module

**Files:** `src/compute/cost.ts` (create)

- [ ] **Step 1:** Create the cost computation function.

```ts
// src/compute/cost.ts

import { pricingFor } from '../pricing/models';

export interface TokenUsage {
  input_tokens: number;
  cache_creation_tokens: number;
  cache_read_tokens: number;
  output_tokens: number;
  model_id: string;
}

export interface CostBreakdown {
  total_usd: number;
  fallback: boolean;
  matched_as: string;
}

export function computeSessionCostUsd(u: TokenUsage): CostBreakdown {
  const { pricing, fallback, matched_as } = pricingFor(u.model_id);
  const cost =
    (u.input_tokens          / 1_000_000) * pricing.input_per_mtok +
    (u.cache_creation_tokens / 1_000_000) * pricing.cache_create_per_mtok +
    (u.cache_read_tokens     / 1_000_000) * pricing.cache_read_per_mtok +
    (u.output_tokens         / 1_000_000) * pricing.output_per_mtok;
  return { total_usd: Number(cost.toFixed(4)), fallback, matched_as };
}
```

```ts
function accumulateTokens(entries: JsonlEntry[]): {
  input: number; cache_create: number; cache_read: number; output: number;
} {
  const seen = new Set<string>();
  let input = 0, cache_create = 0, cache_read = 0, output = 0;
  for (const e of entries) {
    if (e.type !== 'assistant') continue;
    const reqId = e.requestId;
    if (!reqId || seen.has(reqId)) continue;
    seen.add(reqId);
    const u = e.message?.usage;
    if (!u) continue;
    input        += u.input_tokens ?? 0;
    cache_create += u.cache_creation_input_tokens ?? 0;
    cache_read   += u.cache_read_input_tokens ?? 0;
    output       += u.output_tokens ?? 0;
  }
  return { input, cache_create, cache_read, output };
}
```

- [ ] **Step 2:** Add `getTopExpensiveSessions(db, fromMs, n)` to `src/storage/db.ts`:

```ts
export function getTopExpensiveSessions(db, fromMs: number, n: number): TopSession[] {
  return db.prepare(`
    SELECT
      e.session_id,
      e.timestamp,
      ROUND(e.cost_usd, 2) AS cost_usd,
      e.session_end_ms - e.timestamp AS duration_ms,
      COUNT(a.commit_hash) AS attr_count
    FROM events e
    LEFT JOIN attributions a ON a.session_id = e.session_id
    WHERE e.type = 'session' AND e.timestamp >= ?
    GROUP BY e.session_id
    ORDER BY e.cost_usd DESC
    LIMIT ?
  `).all(fromMs, n);
}
```

```ts
function fmtDuration(ms: number): string {
  const minutes = Math.round(ms / 60_000);
  if (minutes < 60) return `${minutes}m`;

  const m = minutes % 60;
  return `${h}h${m.toString().padStart(2, '0')}m`;
```
